typedef struct {
    void *weight;
    float *scale;
    int rows;
    int cols;
} fp8_tensor;

static float fp8_to_float(unsigned char x) {
    int sign = (x >> 7) & 1;
    int exp = (x >> 3) & 15;
    int mant = x & 7;
    float value;
    if (exp == 0) {
        value = mant == 0 ? 0.0f : (float)mant / 512.0f;
    } else {
        value = (1.0f + (float)mant / 8.0f);
        for (int i = 0; i < exp - 7; ++i) value *= 2.0f;
        for (int i = 0; i < 7 - exp; ++i) value *= 0.5f;
    }
    return sign ? -value : value;
}

static void matvec_block_scaled(float *y, const fp8_tensor *t, const float *x) {
    const unsigned char *w = (const unsigned char *)t->weight;
    int scale_cols = (t->cols + 127) / 128;
    for (int r = 0; r < t->rows; ++r) {
        double acc = 0.0;
        for (int c = 0; c < t->cols; ++c) {
            float s = t->scale[(r / 128) * scale_cols + c / 128];
            acc += (double)fp8_to_float(w[(size_t)r * t->cols + c]) * s * x[c];
        }
        y[r] = (float)acc;
    }
}

static int greedy_loop(const fp8_tensor *head, float *state, int max_new) {
    float logits[4096];
    int last = 0;
    for (int step = 0; step < max_new; ++step) {
        matvec_block_scaled(logits, head, state);
        int best = 0;
        for (int i = 1; i < head->rows; ++i) {
            if (logits[i] > logits[best]) best = i;
        }
        last = best;
        for (int i = 0; i < head->cols; ++i) {
            state[i] = 0.99f * state[i] + 0.0001f * (float)(best % 97);
        }
    }
    return last;
}
